<script type="module">
  const {enterFullscreen, exitFullscreen, resizeFontSize, initDataView, isFullScreen} = window.JsToolkit

  console.log('--resizeFontSize--', resizeFontSize(16, 0))

  const myDataView = initDataView(document.getElementById('app-main'), {
    width: 1920, // 设计稿宽度
    height: 1080, // 设计稿高度
    // mode: "scaleToFill",
    mode: "aspectFit",
  })
  window.onresize = function () {
    console.log('--resizeFontSize--', resizeFontSize())
    myDataView.resize()
  }
  console.log(isFullScreen())
  document.getElementById('btn1').onclick = function () {
    enterFullscreen(document.getElementById('body')).then(() => {
      console.log('进入全屏-成功',)
    }).catch(() => {
      console.log('进入全屏-失败')
    })
  }
  document.getElementById('btn2').onclick = function () {
    exitFullscreen().then(() => {
      console.log('退出全屏-成功')
    }).catch(() => {
      console.log('退出全屏-失败')
    })
  }
</script>

enterFullscreen

  • 说明：

    使元素进入全屏模式

  • 添加版本：1.1.0-beta.15

  • 参数：

    • {Element} [element=document.documentElement] - 要进入全屏的元素，默认为整个文档

  • 返回值：

    {Promise<void>} 全屏操作的 Promise

  • 示例：

enterFullscreen()

exitFullscreen

  • 说明：

    退出全屏

  • 添加版本：1.1.0-beta.15

  • 返回值：

    {Promise<void>} 全屏操作的 Promise

  • 示例：

exitFullscreen()

resizeFontSize

  • 说明：

    根据窗口大小自适应字体大小

  • 参数：

    • {number} [fontSize=16] - 初始字体大小
    • {number} [initWidth=1920] - 初始宽度

  • 返回值：

    {number} 返回计算后字体大小

  • 示例：

    参考上面的综合示例代码

initDataView

  • 说明：

    初始化数据可视化容器（用于数据可视化大屏）

  • 添加版本：1.1.0-beta.8

  • 参数：

    • {HTMLElement} el 元素
    • {Object} options 参数 {width, height,mode}
    • {number} options.width 标准/设计稿/实际宽度
    • {number} options.height 标准/设计稿/实际高度
    • {string} options.mode 缩放模式(scaleToFill：拉满全屏缩放(默认), aspectFit：等比缩放)

  • 返回值：

    {DataView} 返回DataView实例对象

  • 示例：

    参考上面的综合示例代码
